355  inline EGS_Float sampleEnergy(EGS_RandomGenerator *rndm) {
356  EGS_Float e = sample(rndm);
357  count++;
358  sum_E += e;
359  sum_E2 += e*e;
360  return e;
361  };

389  virtual bool storeState(ostream &data_out) const {
390  if (!egsStoreI64(data_out,count)) {
391  return false;
392  }
393  data_out << " " << sum_E << " " << sum_E2 << endl;
394  if (!data_out.good() || data_out.fail()) {
395  return false;
396  }
397  return true;
398  };
399 
413  virtual bool setState(istream &data_in) {
414  if (!egsGetI64(data_in,count)) {
415  return false;
416  }
417  data_in >> sum_E >> sum_E2;
418  if (data_in.eof() || !data_in.good() || data_in.fail()) {
419  return false;
420  }
421  return true;
422  };
423 
436  virtual bool addState(istream &data_in) {
437  EGS_I64 count_save = count;
438  double sum_E_save = sum_E, sum_E2_save = sum_E2;
439  if (!setState(data_in)) {
440  return false;
441  }
442  count += count_save;
443  sum_E += sum_E_save;
444  sum_E2 += sum_E2_save;
445  return true;
446  };
447 
459  virtual void resetCounter() {
460  count = 0;
461  sum_E = 0;
462  sum_E2 = 0;
463  };

485  void getSampledAverage(EGS_Float &e, EGS_Float &de) const {
486  if (count > 1) {
487  e = sum_E/count;
488  de = sum_E2/count;
489  de -= e*e;
490  if (de > 0) {
491  de = sqrt(de/(count-1));
492  }
493  }

805 template <class T>
806 EGS_BaseSource *createSourceTemplate(EGS_Input *input,
807  EGS_ObjectFactory *f, const char *name) {
808  EGS_BaseSource::addKnownTypeId(typeid(T).name());
809  if (!input) {
810  egsWarning("createSource(%s): null input?\n",name);
811  return 0;
812  }
813  T *res = new T(input,f);
814  if (!res->isValid()) {
815  egsWarning("createSource(%s): the input is not "
816  "sufficient to create a valid source\n",name);
817  delete res;
818  return 0;
819  }
