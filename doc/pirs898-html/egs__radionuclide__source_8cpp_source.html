312 EGS_I64 EGS_RadionuclideSource::getNextParticle(EGS_RandomGenerator *rndm, int
313  &q, int &latch, EGS_Float &E, EGS_Float &wt, EGS_Vector &x, EGS_Vector
314  &u) {
315 
316  unsigned int i = 0;
317  if (decays.size() > 1 && disintegrationOccurred) {
318  // Sample a uniform random number
319  EGS_Float uRand = rndm->getUniform();
320 
321  // Sample which spectrum to use
322  for (i=0; i<decays.size(); ++i) {
323  if (uRand < decays[i]->getSpectrumWeight()) {
324  break;
325  }
326  }
327  }
328 
329  // Check if the emission time is within the experiment time limit
330  if (experimentTime <= 0. || time < experimentTime) {
331  // Keep this particle
332  }
333  else {
334  // If the particle was emitted outside the
335  // experiment time window, just set the energy to zero to discard
336  E = 0;
337  return ++count;
338  }
339 
340  EGS_I64 ishowerOld = decays[i]->getShowerIndex();
341 
342  E = decays[i]->sampleEnergy(rndm);
343 
344  EGS_I64 ishowerNew = decays[i]->getShowerIndex();
345  if (ishowerNew > ishowerOld) {
346  disintegrationOccurred = true;
347  time = lastDisintTime + -log(1.-rndm->getUniform()) / activity * (ishowerNew - ishowerOld);
348 
349  lastDisintTime = time;
350  ishower += (ishowerNew - ishowerOld);
351  }
352  else {
353  disintegrationOccurred = false;
354  // The time returned from the spectrum is just the time
355  // since the last disintegration event, so this is only
356  // non-zero for internal transitions. This is why the
357  // times are added here - each transition occurs only after
358  // the delay of the previous.
359  time += decays[i]->getTime();
360  }
361 
362  q = decays[i]->getCharge();
363 
364  getPositionDirection(rndm,x,u,wt);
365  latch = 0;
366 
367  if (disintegrationOccurred) {
368  xOfDisintegration = x;
369  }
370  else {
371  x = xOfDisintegration;
372  }
373 
374  // Now that we have the position of the particle, we can
375  // find the region and deposit any sub-threshold contributions locally
376  // This includes edep from relaxations, and alpha particle energy
377  EGS_Float edep = decays[i]->getEdep();
378  if (edep > 0) {
379  app->setEdep(edep);
380  int ireg = app->isWhere(x);
381  app->userScoring(3, ireg);
382  }
383 
384  // Check if the charge is allowed
385  if (q_allowAll || std::find(q_allowed.begin(), q_allowed.end(), q) != q_allowed.end()) {
386  // Keep the particle
387  }
388  else {
389  // Don't transport the particle
390  E = 0;
391  }
392 
393  // If the energy is zero, also set the weight to zero
394  // If you don't do this, electrons will still be given their rest
395  // mass energy in shower()
396  if (E < epsilon) {
397  wt = 0;
398  }
399 
400  return ++count;
401 }

403 void EGS_RadionuclideSource::getPositionDirection(EGS_RandomGenerator *rndm,
404  EGS_Vector &x, EGS_Vector &u, EGS_Float &wt) {
405 
406  if (sourceType == "isotropic") {
407  bool ok = true;
408  do {
409  x = shape->getRandomPoint(rndm);
410  if (geom) {
411  if (gc == IncludeAll) {
412  ok = geom->isInside(x);
413  }
414  else if (gc == ExcludeAll) {
415  ok = !geom->isInside(x);
416  }
417  else if (gc == IncludeSelected) {
418  ok = false;
419  int ireg = geom->isWhere(x);
420  for (int j=0; j<nrs; ++j) {
421  if (ireg == regions[j]) {
422  ok = true;
423  break;
424  }
425  }
426  }
427  else {
428  ok = true;
429  int ireg = geom->isWhere(x);
430  for (int j=0; j<nrs; ++j) {
431  if (ireg == regions[j]) {
432  ok = false;
433  break;
434  }
435  }
436  }
437  }
438  }
439  while (!ok);
440  u.z = buf_1 - rndm->getUniform()*(buf_1 - buf_2);
441  EGS_Float sinz = 1-u.z*u.z;
442  if (sinz > epsilon) {
443  sinz = sqrt(sinz);
444  EGS_Float cphi, sphi;
445  EGS_Float phi = min_phi +(max_phi - min_phi)*rndm->getUniform();
446  cphi = cos(phi);
447  sphi = sin(phi);
448  u.x = sinz*cphi;
449  u.y = sinz*sphi;
450  }
451  else {
452  u.x = 0;
453  u.y = 0;
454  }
455  wt = 1;
456  }
457  else if (sourceType == "collimated") {
458  x = source_shape->getRandomPoint(rndm);
459  int ntry = 0;
460  do {
461  target_shape->getPointSourceDirection(x,rndm,u,wt);
462  ntry++;
463  if (ntry > 10000) {
464  egsFatal("EGS_RadionuclideSource::getPositionDirection:\n"
465  " my target shape %s, which is of type %s, failed to\n"
466  " return a positive weight after 10000 attempts\n",
467  target_shape->getObjectName().c_str(),
468  target_shape->getObjectType().c_str());
469  }
470  }
471  while (wt <= 0);
472 
473  if (disintegrationOccurred) {
474  ctry += ntry-1;
475  }
476  }
